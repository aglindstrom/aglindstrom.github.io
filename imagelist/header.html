#ifndef _IMAGELIST_H_
#define _IMAGELIST_H_

#include <string.h>

class Imagelist
{
	struct imageNode{
		char* filename;
		unsigned char* texture;
		unsigned int index;
		int width, height, size;
		
		imageNode* next;
		imageNode* prev;
	};

	public:
		Imagelist();
		Imagelist(const Imagelist&);
		~Imagelist();

		bool Initialize(char*);
		bool addTexture(char*);
		void Shutdown();

		void getFileName(char*);
		void getImage(unsigned char*, int);
		int getWidth(){ return first->width; }
		int getHeight(){ return first->height; }
		int getSize(){ return first->size; }
		int getIndex(){ return first->index; }

		void setCurrent(int);
		void setImage(unsigned char*, int);
		void setWidth(int width){ first->width = width; }
		void setHeight(int height){ first->height = height; }
		void setSize(int size){ first->size = size; }
		int lastIndex();
		void getNext() { current = current->next; }

		bool newImage(int size){ current->texture = new unsigned char[size]; if (!current->texture){ return false; } return true; }

	private:

		imageNode* first;
		imageNode* last;
		imageNode* current;

		void deleteNode(int);
		bool addNode(char*);
		void modifyImage(unsigned char*, int);
};

#endif
